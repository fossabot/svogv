Edit current user
Techniques used

This form makes use of the <ac-editor> component.

 <form (ngSubmit)="saveUser()" [formGroup]="myForm" role="form" class="row">
    <fieldset>
      <legend>Edit current user</legend>
      <ac-autoform [userForm]="myForm"></ac-editor>

The form needs a name, here we used 'myForm'. The component AcAutoForm reads the associated view model and creates all the fields including validation, label, and data type specific information. 

export class UserViewModel {{ '{' }}

  @Hidden()
  id: number = 0;

  @Display('E-Mail', 30, 'E-Mail address')
  @Required()
  @MaxLength(100)
  @Email()
  email: string = '';

  @Display('Phone Number', 20, 'The user\'s phone')
  @Required()
  @MaxLength(20)
  phoneNumber: string = '';

  @Display('User Name', 10, 'The full name')
  @Required()
  @MaxLength(100)
  userName: string = '';

  @Display('Age', 40, 'From 12 to 88')
  @Range(12, 88)
  age = 14;

  @Display('Password', 50)
  @Required()
  @Compare('passWordTwo')
  passWord: string = '';

  @Display('Password', 60)
  @Required()
  passWordTwo: string = '';

{{ '}' }}
